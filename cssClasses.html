//--- LoadContentPages ------------------------------------

function LoadContentPages ()
{
  try
  {
    const contentPages = $(".contentPage").toArray ();
    const totalPages   = contentPages.length;

    // Load all pages recursively to keep them in order
    if (totalPages > 0)
    {
      let pageIndex = 0;
      loadPage ();

      function loadPage ()
      {
        if (pageIndex < totalPages)
        {
          const pageID = contentPages[pageIndex++].id;
          $("#" + pageID).load (pageID + ".html", loadPage);
        }
        else
        {
          // Show first page
          $(".navButton").first().click();
        }
      }
    }
  }
  catch (ex)
  {
    ShowException (ex);
  }
}
